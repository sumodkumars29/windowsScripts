where windhawk
dir C:\*windhawk* /s /a
dir C:\*windhawk* /s /b /a
dir "C:\Program Files\*windhawk*" /s /b /a
dir "C:\Users\%USERNAME%\AppData\*windhawk*" /s /b /a

# *windhawk* → matches any filename containing “windhawk”.
# /s → recursive (all subdirectories).
# /a → show hidden/system files too.
# /b = bare format, only full paths)

# Searcing in registries
reg query HKCU\Software | findstr /i windhawk
reg query HKLM\Software | findstr /i windhawk
reg query HKLM\SOFTWARE\Microsoft\Windows\CurrentVersion\Uninstall /s | findstr /i windhawk
reg query HKLM\SOFTWARE\WOW6432Node\Microsoft\Windows\CurrentVersion\Uninstall /s | findstr /i windhawk
